                                                    <h2>
                                                        H. A. Rey</h2>
                                                    <p>
                                                        Hans Augusto Rey was born in Hamburg, Germany in 1898. As a child, he spent much of his free time in that city's famous Hagenbeck Zoo drawing animals. After serving in the army during World War I, he studied philology and natural science at the University of Hamburg. He then married Margret Rey and they moved to Montmartre for four years. The manuscript for the first Curious George books was one of the few items the Reys carried with them on their bicycles when they escaped from Paris in 1940. Eventually, they made their way to the United States, and Curious George was published in 1941. Curious George has been published in many languages, including French, German, Japanese, Afrikaans, and Norwegian. Additional Curious George books followed, as well as such other favorites as CECILY G. AND THE NINE MONKEYS and FIND THE CONSTELLATIONS.

<script>

    equalheight = function (container) {

        var currentTallest = 0,
             currentRowStart = 0,
             rowDivs = new Array(),
             $el,
             topPosition = 0;
        $(container).each(function () {

            $el = $(this);
            $($el).height('auto')
            topPostion = $el.position().top;

            if (currentRowStart != topPostion) {
                for (currentDiv = 0 ; currentDiv < rowDivs.length ; currentDiv++) {
                    rowDivs[currentDiv].height(currentTallest);
                }
                rowDivs.length = 0; // empty the array
                currentRowStart = topPostion;

            var itemText = item.text();

            if (itemAttr == 't.t2') //category
            {
                itemText = itemText.substring(0, itemText.indexOf('(') - 1);
                oItemClick.prop49 = itemText;
            }
            if (itemAttr == 't.t3') //sub-category
            {
                itemText = itemText.substring(0, itemText.indexOf('(') - 1);
                oItemClick.prop50 = itemText;
            }
            else if (itemAttr == 'topic') {
                itemText = itemText.substring(0, itemText.indexOf('(') - 1);
                oItemClick.prop51 = itemText;
            }

            if (itemAttr == 'paging') {
                oItemClick.pageName = 'Search Results:' + item.text();
                //oItemClick.linkTrackVars += ",pageName";
                //oItemClick.tl(true, 'o', 'PageName: ' + item.text());
            }
            else if (itemAttr == 'pagingprev' || itemAttr == 'pagingnext') {
                itemSearch = item[0].search;
                var items = itemSearch.split(';');
                var pageNumber = '';
                var hasPageNumber = false;
                for (var i = 0; i < items.length; i++) {
                    var kvp = items[i].split('=');
                    if (kvp[0] == "page") {
                        pageNumber = kvp[1];
                        hasPageNumber = true;
                        break;
                    }
                }
                if (hasPageNumber == false)
                    pageNumber = 1;

                oItemClick.pageName = ' Search Results:' + pageNumber;
                oItemClick.linkTrackVars += ",pageName";
                oItemClick.tl(true, 'o', 'PageName: ' + pageNumber);
            }

            if (DEBUG_CLICK) {
                alert('Item Click: ' + item.text());
            }
        }
