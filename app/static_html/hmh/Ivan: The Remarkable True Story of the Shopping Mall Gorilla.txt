Brian Karas present the extraordinary real story of a special gorilla.<BR>&#160;&#160;&#160;&#160; Captured as a baby, Ivan was brought to a Tacoma, Washington, mall to attract shoppers. Gradually, public pressure built until a better way of life for Ivan was found at Zoo Atlanta. From the Congo to America, and from a local business attraction to a national symbol of animal welfare, Ivan the Shopping Mall Gorilla traveled an astonishing distance in miles and in impact.<BR>&#160;&#160;&#160;&#160; This is his true story and includes photographs of Ivan in the back matter.</DIV>

A heartfelt tribute to a magnificent animal.&rdquo;&nbsp;<p><B>&mdash;Shelf Awareness, starred review</B>&nbsp;<p>&nbsp;<p> &quot;The text is streamlined but powerful, and younger kids who aren&rsquo;t up for the linguistic and emotional challenges of Applegate&rsquo;s fictional account of Ivan will find this a thoughtful and intriguing introduction; older readers who have read The One and Only Ivan may enjoy it as well.&quot;&nbsp;<p><I><B>&mdash;Bulletin</B></I></DIV>

    equalheight = function (container) {

        var currentTallest = 0,
             currentRowStart = 0,
             rowDivs = new Array(),
             $el,
             topPosition = 0;
        $(container).each(function () {

            $el = $(this);
            $($el).height('auto')
            topPostion = $el.position().top;

            if (currentRowStart != topPostion) {
                for (currentDiv = 0 ; currentDiv < rowDivs.length ; currentDiv++) {
                    rowDivs[currentDiv].height(currentTallest);
                }
                rowDivs.length = 0; // empty the array
                currentRowStart = topPostion;
                currentTallest = $el.height();
                rowDivs.push($el);
            } else {
                rowDivs.push($el);
                currentTallest = (currentTallest < $el.height()) ? ($el.height()) : (currentTallest);
            }
            for (currentDiv = 0 ; currentDiv < rowDivs.length ; currentDiv++) {
                rowDivs[currentDiv].height(currentTallest);
            }
        });
    }

        function Om_ItemClick(item) {
            var oItemClick = GetOmObj();
            var itemAttr = item.attr('data-section');
            var itemText = item.text();

            if (itemAttr == 't.t2') //category
            {
                itemText = itemText.substring(0, itemText.indexOf('(') - 1);
                oItemClick.prop49 = itemText;
            }
            if (itemAttr == 't.t3') //sub-category
            {
                itemText = itemText.substring(0, itemText.indexOf('(') - 1);
                oItemClick.prop50 = itemText;
            }
            else if (itemAttr == 'topic') {
                itemText = itemText.substring(0, itemText.indexOf('(') - 1);
                oItemClick.prop51 = itemText;
            }

            if (itemAttr == 'paging') {
                oItemClick.pageName = 'Search Results:' + item.text();
                //oItemClick.linkTrackVars += ",pageName";
                //oItemClick.tl(true, 'o', 'PageName: ' + item.text());
            }
            else if (itemAttr == 'pagingprev' || itemAttr == 'pagingnext') {
                itemSearch = item[0].search;
                var items = itemSearch.split(';');
                var pageNumber = '';
                var hasPageNumber = false;
                for (var i = 0; i < items.length; i++) {
                    var kvp = items[i].split('=');
                    if (kvp[0] == "page") {
                        pageNumber = kvp[1];
                        hasPageNumber = true;
                        break;
                    }
                }
                if (hasPageNumber == false)
                    pageNumber = 1;

                oItemClick.pageName = ' Search Results:' + pageNumber;
                oItemClick.linkTrackVars += ",pageName";
                oItemClick.tl(true, 'o', 'PageName: ' + pageNumber);
            }

            if (DEBUG_CLICK) {
                alert('Item Click: ' + item.text());
            }
        }
