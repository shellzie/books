                                                    <h2>
                                                        H. A. Rey</h2>
                                                    <p>
                                                        The Reys were born in Hamburg, Germany. Hans Augusto Rey (1898-1977) met his wife-to-be, Margret (1906-1996), at a party in her father&rsquo;s home in Germany; when he first caught a glimpse of her, she was sliding down the banister. In their twenties and thirties they lived in Paris and in Rio de Janeiro, where Hans sold bathtubs in villages along the Amazon River. Eventually Cambridge, Massachusetts, became the Reys&rsquo; home and community. Throughout their lives the Reys created many lively books together, including SPOTTY, PRETZEL, and lift-the-flap books such as HOW DO YOU GET THERE? The manuscript of the first Curious George books was one of the few items the Reys carried with them on their bicycles when they escaped from Paris in 1940. Eventually, they made their way to the United States, and CURIOUS GEORGE was published in 1941. Their incorrigible little monkey has become an American icon, selling millions of books and capturing the hearts of readers everywhere. CURIOUS GEORGE has been published in many languages, including French, German, Japanese, Afrikaans, and Norwegian. Additional Curious George books followed, as well as such other favorites as CECILY G. AND THE NINE MONKEYS and FIND THE CONSTELLATIONS.

    equalheight = function (container) {

        var currentTallest = 0,
             currentRowStart = 0,
             rowDivs = new Array(),
             $el,
             topPosition = 0;
        $(container).each(function () {

            $el = $(this);
            $($el).height('auto')
            topPostion = $el.position().top;

            if (currentRowStart != topPostion) {
                for (currentDiv = 0 ; currentDiv < rowDivs.length ; currentDiv++) {
                    rowDivs[currentDiv].height(currentTallest);
                }
                rowDivs.length = 0; // empty the array
                currentRowStart = topPostion;
                currentTallest = $el.height();
                rowDivs.push($el);
            } else {
                rowDivs.push($el);
                currentTallest = (currentTallest < $el.height()) ? ($el.height()) : (currentTallest);
            }
            for (currentDiv = 0 ; currentDiv < rowDivs.length ; currentDiv++) {
                rowDivs[currentDiv].height(currentTallest);
            }
        });
    }

        function Om_CartFunction(action, isbn) {

            if (action == 'checkout') {
                oCart.events = 'scCheckout';

                for (var i = 0; i < currentCartItems.length; i++) {
                    if (i == 0) {
                        oCart.products = ";" + currentCartItems[i];

                        oCart.products += ",;" + currentCartItems[i];

                }

                oCart.events = 'scRemove';
                oCart.linkTrackEvents = 'scRemove';

            else if (action == 'viewcart') {
                oCart.events = 'scView';
                oCart.linkTrackEvents = 'scView';

            if (DEBUG_CLICK)
                alert('Cart Function Click (' + action + ') complete');

        function Om_ItemClick(item) {
            var oItemClick = GetOmObj();
            var itemAttr = item.attr('data-section');
            var itemText = item.text();

            if (itemAttr == 't.t2') //category
            {
                itemText = itemText.substring(0, itemText.indexOf('(') - 1);
                oItemClick.prop49 = itemText;
            }
            if (itemAttr == 't.t3') //sub-category
            {
                itemText = itemText.substring(0, itemText.indexOf('(') - 1);
                oItemClick.prop50 = itemText;
            }
            else if (itemAttr == 'topic') {
                itemText = itemText.substring(0, itemText.indexOf('(') - 1);
                oItemClick.prop51 = itemText;
            }

            if (itemAttr == 'paging') {
                oItemClick.pageName = 'Search Results:' + item.text();
                //oItemClick.linkTrackVars += ",pageName";
                //oItemClick.tl(true, 'o', 'PageName: ' + item.text());
            }
            else if (itemAttr == 'pagingprev' || itemAttr == 'pagingnext') {
                itemSearch = item[0].search;
                var items = itemSearch.split(';');
                var pageNumber = '';
                var hasPageNumber = false;
                for (var i = 0; i < items.length; i++) {
                    var kvp = items[i].split('=');
                    if (kvp[0] == "page") {
                        pageNumber = kvp[1];
                        hasPageNumber = true;
                        break;
                    }
                }
                if (hasPageNumber == false)
                    pageNumber = 1;

                oItemClick.pageName = ' Search Results:' + pageNumber;
                oItemClick.linkTrackVars += ",pageName";
                oItemClick.tl(true, 'o', 'PageName: ' + pageNumber);
            }

            if (DEBUG_CLICK) {
                alert('Item Click: ' + item.text());
            }
        }
