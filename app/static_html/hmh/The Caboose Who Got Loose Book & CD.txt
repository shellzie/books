                                                    <h2>
                                                        Bill Peet</h2>
                                                    <p>
                                                        Bill Peet was the author of 34 books published by Houghton Mifflin. One of these, BILL PEET: AN AUTOBIOGRAPHY, was named a 1989 Caldecott Honor Book. All of Bill Peet&rsquo;s books published by Houghton Mifflin Company, including his first book for children published in 1959, HUBERT'S HAIR-RAISING ADVENTURE, remain actively in print today. 

In both his career as an author and illustrator of children&rsquo;s books and in his work as sketch artist and continuity illustrator at Walt Disney, Bill Peet created a menagerie of memorable characters. As he himself noted, &quot;I write about animals because I love to draw them. Most of my animal characters have human personalities, and some are much like the people I know.&quot;

At Walt Disney, where Bill Peet worked for 27 years, he was a key participant in the production of classic films such as Fantasia, Sleeping Beauty, Alice in Wonderland, Peter Pan, and 101 Dalmatians for which he was not only an artist, but the screenwriter as well. 

Bill Peet&rsquo;s signature style enabled him to create fast-paced stories of fantastical adventure delivered with warmth and laugh-out-loud hilarity. His unfailing humor did not, however, prevent him from addressing such poignant issues as kindness toward others and respect for the environment. Through the exploits of his characters, Peet offered his audience a chance to see themselves and their world through new eyes.

&quot;At some point,&quot; Bill Peet once said, &quot;it occurred to me that drawing was something I couldn&rsquo;t possibly give up, and somehow it must be turned into a profession.&quot; He went on to not only fulfill his dream but to introduce generations of young readers to his delightful vision of humor, friendship and compassion.

    equalheight = function (container) {

        var currentTallest = 0,
             currentRowStart = 0,
             rowDivs = new Array(),
             $el,
             topPosition = 0;
        $(container).each(function () {

            $el = $(this);
            $($el).height('auto')
            topPostion = $el.position().top;

            if (currentRowStart != topPostion) {
                for (currentDiv = 0 ; currentDiv < rowDivs.length ; currentDiv++) {
                    rowDivs[currentDiv].height(currentTallest);
                }
                rowDivs.length = 0; // empty the array
                currentRowStart = topPostion;
                currentTallest = $el.height();
                rowDivs.push($el);
            } else {
                rowDivs.push($el);
                currentTallest = (currentTallest < $el.height()) ? ($el.height()) : (currentTallest);
            }
            for (currentDiv = 0 ; currentDiv < rowDivs.length ; currentDiv++) {
                rowDivs[currentDiv].height(currentTallest);
            }
        });
    }

        function Om_CartFunction(action, isbn) {

            oCart.prop19 = action;
            oCart.linkTrackVars += ",prop19,products";

            //By default, we need to set the ISBN here. It may get overridden at a later place
            if (isbn != '')
                oCart.products = ';' + isbn;

            if (action == 'checkout') {
                oCart.events = 'scCheckout';
                oCart.linkTrackEvents = 'scCheckout';

                for (var i = 0; i < currentCartItems.length; i++) {
                    if (i == 0) {
                        oCart.products = ";" + currentCartItems[i];
                    }
                    else {
                        oCart.products += ",;" + currentCartItems[i];
                    }
                }
            }
            else if (action == 'remove') {
                oCart.events = 'scRemove';
                oCart.linkTrackEvents = 'scRemove';
            }
            else if (action == 'viewcart') {
                oCart.events = 'scView';
                oCart.linkTrackEvents = 'scView';
            }

            oCart.tl(true, 'o', 'Cart Function: ' + action);

            if (DEBUG_CLICK)
                alert('Cart Function Click (' + action + ') complete');
        }

        function Om_ItemClick(item) {
            var oItemClick = GetOmObj();
            var itemAttr = item.attr('data-section');
            var itemText = item.text();

            if (itemAttr == 't.t2') //category
            {
                itemText = itemText.substring(0, itemText.indexOf('(') - 1);
                oItemClick.prop49 = itemText;
            }
            if (itemAttr == 't.t3') //sub-category
            {
                itemText = itemText.substring(0, itemText.indexOf('(') - 1);
                oItemClick.prop50 = itemText;
            }
            else if (itemAttr == 'topic') {
                itemText = itemText.substring(0, itemText.indexOf('(') - 1);
                oItemClick.prop51 = itemText;
            }

            if (itemAttr == 'paging') {
                oItemClick.pageName = 'Search Results:' + item.text();
                //oItemClick.linkTrackVars += ",pageName";
                //oItemClick.tl(true, 'o', 'PageName: ' + item.text());
            }
            else if (itemAttr == 'pagingprev' || itemAttr == 'pagingnext') {
                itemSearch = item[0].search;
                var items = itemSearch.split(';');
                var pageNumber = '';
                var hasPageNumber = false;
                for (var i = 0; i < items.length; i++) {
                    var kvp = items[i].split('=');
                    if (kvp[0] == "page") {
                        pageNumber = kvp[1];
                        hasPageNumber = true;
                        break;
                    }
                }
                if (hasPageNumber == false)
                    pageNumber = 1;

                oItemClick.pageName = ' Search Results:' + pageNumber;
                oItemClick.linkTrackVars += ",pageName";
                oItemClick.tl(true, 'o', 'PageName: ' + pageNumber);
            }

            if (DEBUG_CLICK) {
                alert('Item Click: ' + item.text());
            }
        }
